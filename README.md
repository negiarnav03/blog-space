# first install the required technologies/dependencies.

bun create vite@latest

bun add @reduxjs/toolkit react-redux

bun add react-router-dom

bun add appwrite

bun add @tinymce/tinymce-react

bun add html-react-parser

bun add react-hook-form

# to run the project use
# this will run the project on default port 
bun vite   

OR   
# this will run the project on port 5173
bun run dev
